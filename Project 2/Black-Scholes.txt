Sub Button1_Click()Dim S0, K, r, q, sigma, TS0 = Cells(1, 2)K = Cells(2, 2)r = Cells(3, 2)q = Cells(4, 2)sigma = Cells(5, 2)T = Cells(6, 2)Dim d1, d2d1 = (Log(S0 / K) + (r - q + (sigma * sigma) / 2) * T) / (sigma * Sqr(T))d2 = d1 - sigma * Sqr(T)Dim callprice, putpricecallprice = S0 * (Exp(-q * T)) * WorksheetFunction.NormSDist(d1) - K * (Exp(-r * T)) * WorksheetFunction.NormSDist(d2)putprice = K * (Exp(-r * T)) * WorksheetFunction.NormSDist(-d2) - S0 * (Exp(-q * T)) * WorksheetFunction.NormSDist(-d1)Cells(2, 6) = "Call Price"Cells(3, 6) = "Put Price"Cells(2, 7) = callpriceCells(3, 7) = putpriceEnd Sub
